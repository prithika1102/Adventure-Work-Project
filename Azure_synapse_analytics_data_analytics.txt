----------------------------------------------------------------------------------------------------------------
----------- create master key for the external table
CREATE MASTER Key ENCRYPTION BY PASSWORD = '<password>'

------------ create credential to access the data from data lake
CREATE DATABASE SCOPED CREDENTIAL <credential name>
WITH 
IDENTITY = 'Managed Identity'

------------- create data source for a silver container
CREATE EXTERNAL DATA SOURCE source_silver --- source_silver is the url (data source) for silver container
WITH
(
    LOCATION ='https://<storage-account>.blob.core.windows.net/silver',
    CREDENTIAL = <credential name>
)

------------- create data source for a gold container
CREATE EXTERNAL DATA SOURCE source_gold
WITH
(
    LOCATION ='https://<storage-account>.blob.core.windows.net/gold',
    CREDENTIAL = <credential name>
)

------------ create file format
CREATE EXTERNAL FILE FORMAT format_parquet
WITH
(
    FORMAT_TYPE = PARQUET,
    DATA_COMPRESSION = 'org.apache.hadoop.io.compress.SnappyCodec'
)


-------------------------------------------------------------------------------------------


-------------- create external table

CREATE EXTERNAL TABLE gold.extcalendar
WITH
(
    LOCATION = 'extcalendar',
    DATA_SOURCE = source_gold,
    FILE_FORMAT = format_parquet
) As 
SELECT * FROM gold.calendar
